
cmake_minimum_required(VERSION 2.8)

project({{project_name}})
set(PROJECT_VERSION "0.0.0.0")
set(PROJECT_BRIEF "\"A C++ CMake template\"")

include(${CMAKE_SOURCE_DIR}/cmake/Include.cmake)

# Add required packages
find_package(Boost COMPONENTS program_options filesystem thread system random REQUIRED)
find_package(ROOT REQUIRED)

# Add include directories
include_directories(
    ${Boost_INCLUDE_DIR}
    ${ROOT_INCLUDE_DIR}
    )

# Add libraries
set(LIBRARIES
    ${Boost_LIBRARIES}
    ${ROOT_LIBRARIES}
    )

# Enable/disable testing
option(TESTS_ENABLED "Enable automatic tests" ON)
if(TESTS_ENABLED)
    add_subdirectory(test)
endif()

# Continue in src directory
add_subdirectory(src)
